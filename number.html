<h2>Wpisz numer biletu</h2>


<form id="ticketForm">

    <input type="text" id="ticketNumber" placeholder="Wpisz numer" required>

    <button type="submit">Przejdź do biletu</button>

</form>


<script>

document.getElementById('ticketForm').addEventListener('submit', function(e) {

    e.preventDefault(); // zapobiega przeładowaniu strony


    const ticketNumber = document.getElementById('ticketNumber').value.trim();

    if (ticketNumber) {

        // przekierowanie na stronę /ticket?id=…

        window.location.href = `/ticket?id=${encodeURIComponent(ticketNumber)}`;

    } else {

        alert("Wpisz numer biletu!");

    }

});

</script>
